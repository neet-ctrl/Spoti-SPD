We'll add a new "Library" tab to the bottom bar that automatically scans and displays all locally downloaded music files with rich metadata, total counts, and seamless integration with the existing media player. Below is a comprehensive implementation guide covering architecture, code, animations, and advanced features.

---

🎯 Feature Overview

· Auto‑scan internal storage for audio files (MP3, M4A, FLAC, etc.) from the configured download directory and any user‑selected folders.
· Metadata extraction using MediaMetadataRetriever: title, artist, album, duration, cover art (embedded or from folder.jpg).
· Smart caching with Room database for instant load times after first scan.
· New "Library" tab in bottom navigation with:
  · Total song count, total duration, and storage used.
  · Beautiful Material 3 list with animated items, shimmer while loading, and smooth transitions.
  · Sorting & filtering (by title, artist, album, recently added).
  · Swipe‑to‑play / swipe‑to‑delete actions.
· Seamless playback – tapping any song starts playback via the existing MusicPlayerService and shows the floating player.
· Extra features: create custom playlists, mark favorites, and sync with the existing download history.

🔍 3. Scanning Local Files

We'll create a LibraryScanner utility that runs in a background coroutine and updates the database.

3.1. Permission Handling

The app already requests storage permissions for downloads. We'll reuse the existing Permission.kt to ensure READ_EXTERNAL_STORAGE is granted before scanning.

3.2. Scanning Logic

```kotlin
// utils/LibraryScanner.kt
class LibraryScanner(
    private val context: Context,
    private val dao: LocalSongDao
) {
    suspend fun scanAndUpdate(onProgress: (Int, Int) -> Unit = { _, _ -> }) {
        withContext(Dispatchers.IO) {
            val downloadDir = File(getDownloadDirectoryPath())
            val audioFiles = mutableListOf<File>()

            // Recursively find audio files
            downloadDir.walkTopDown().forEach { file ->
                if (file.isFile && isAudioFile(file.extension)) {
                    audioFiles.add(file)
                }
            }

            val total = audioFiles.size
            var processed = 0

            val songs = audioFiles.mapNotNull { file ->
                val metadata = extractMetadata(file)
                processed++
                withContext(Dispatchers.Main) { onProgress(processed, total) }
                metadata?.let { createLocalSong(file, it) }
            }

            // Insert all scanned songs (replaces existing ones)
            dao.insertAll(songs)

            // Delete songs that no longer exist
            val existingPaths = songs.map { it.filePath }
            dao.deleteMissing(existingPaths)
        }
    }

    private fun isAudioFile(extension: String): Boolean {
        val lowerExt = extension.lowercase()
        return lowerExt in setOf("mp3", "m4a", "flac", "wav", "ogg", "aac")
    }

    private fun extractMetadata(file: File): MediaMetadata? {
        return try {
            val retriever = MediaMetadataRetriever()
            retriever.setDataSource(file.absolutePath)
            MediaMetadata(
                title = retriever.extractMetadata(MediaMetadataRetriever.METADATA_KEY_TITLE)
                    ?: file.nameWithoutExtension,
                artist = retriever.extractMetadata(MediaMetadataRetriever.METADATA_KEY_ARTIST) ?: "Unknown Artist",
                album = retriever.extractMetadata(MediaMetadataRetriever.METADATA_KEY_ALBUM) ?: "Unknown Album",
                duration = retriever.extractMetadata(MediaMetadataRetriever.METADATA_KEY_DURATION)?.toLongOrNull() ?: 0L
            )
        } catch (e: Exception) {
            e.printStackTrace()
            null
        } finally {
            retriever.release()
        }
    }

    private fun createLocalSong(file: File, metadata: MediaMetadata): LocalSong {
        return LocalSong(
            filePath = file.absolutePath,
            title = metadata.title,
            artist = metadata.artist,
            album = metadata.album,
            duration = metadata.duration,
            size = file.length(),
            dateModified = file.lastModified()
        )
    }

    private fun getDownloadDirectoryPath(): String {
        // Retrieve from SharedPreferences (default: /storage/emulated/0/Download/Spotify Downloads/)
        val prefs = context.getSharedPreferences("app_prefs", Context.MODE_PRIVATE)
        return prefs.getString("download_location", Environment.getExternalStoragePublicDirectory(Environment.DIRECTORY_DOWNLOADS).absolutePath + "/Spotify Downloads") ?: ""
    }

    private data class MediaMetadata(
        val title: String,
        val artist: String,
        val album: String,
        val duration: Long
    )
}
```

🖥️ 4. LibraryScreen UI

4.1. LibraryViewModel

```kotlin
// ui/viewmodel/LibraryViewModel.kt
class LibraryViewModel(
    private val dao: LocalSongDao,
    private val scanner: LibraryScanner
) : ViewModel() {

    private val _isScanning = MutableStateFlow(false)
    val isScanning: StateFlow<Boolean> = _isScanning.asStateFlow()

    private val _scanProgress = MutableStateFlow(Pair(0, 0))
    val scanProgress: StateFlow<Pair<Int, Int>> = _scanProgress.asStateFlow()

    val allSongs: StateFlow<List<LocalSong>> = dao.getAllSongs()
        .stateIn(viewModelScope, SharingStarted.WhileSubscribed(5000), emptyList())

    val songCount: StateFlow<Int> = dao.getSongCount()
        .stateIn(viewModelScope, SharingStarted.WhileSubscribed(5000), 0)

    val totalDuration: StateFlow<Long> = dao.getTotalDuration()
        .stateIn(viewModelScope, SharingStarted.WhileSubscribed(5000), 0L)

    val totalSize: StateFlow<Long> = dao.getTotalSize()
        .stateIn(viewModelScope, SharingStarted.WhileSubscribed(5000), 0L)

    private val _sortOrder = MutableStateFlow(SortOrder.TITLE_ASC)
    val sortedSongs: StateFlow<List<LocalSong>> = combine(allSongs, _sortOrder) { songs, order ->
        when (order) {
            SortOrder.TITLE_ASC -> songs.sortedBy { it.title }
            SortOrder.TITLE_DESC -> songs.sortedByDescending { it.title }
            SortOrder.ARTIST_ASC -> songs.sortedBy { it.artist }
            SortOrder.ARTIST_DESC -> songs.sortedByDescending { it.artist }
            SortOrder.DURATION_DESC -> songs.sortedByDescending { it.duration }
            SortOrder.RECENTLY_ADDED -> songs.sortedByDescending { it.dateModified }
        }
    }.stateIn(viewModelScope, SharingStarted.WhileSubscribed(5000), emptyList())

    fun refreshLibrary() {
        viewModelScope.launch {
            _isScanning.value = true
            scanner.scanAndUpdate { processed, total ->
                _scanProgress.value = Pair(processed, total)
            }
            _isScanning.value = false
        }
    }

    fun updateSortOrder(order: SortOrder) {
        _sortOrder.value = order
    }

    fun toggleFavorite(song: LocalSong) {
        viewModelScope.launch {
            val updated = song.copy(isFavorite = !song.isFavorite)
            dao.update(updated)
        }
    }

    fun deleteSong(song: LocalSong) {
        viewModelScope.launch {
            val file = File(song.filePath)
            if (file.exists()) file.delete()
            dao.delete(song)
        }
    }

    enum class SortOrder {
        TITLE_ASC, TITLE_DESC, ARTIST_ASC, ARTIST_DESC, DURATION_DESC, RECENTLY_ADDED
    }
}
```

```kotlin
// ui/screen/LibraryScreen.kt
@Composable
fun LibraryScreen(
    viewModel: LibraryViewModel = hiltViewModel(),
    onPlaySong: (LocalSong) -> Unit,    // passed from MainActivity to start playback
    playerState: PlayerState?            // from HomeScreenViewModel to show floating player
) {
    val isScanning by viewModel.isScanning.collectAsState()
    val scanProgress by viewModel.scanProgress.collectAsState()
    val songs by viewModel.sortedSongs.collectAsState()
    val songCount by viewModel.songCount.collectAsState()
    val totalDuration by viewModel.totalDuration.collectAsState()
    val totalSize by viewModel.totalSize.collectAsState()

    val sortOrder by viewModel.sortOrder.collectAsState() // if exposed

    var showSortDialog by remember { mutableStateOf(false) }

    Scaffold(
        topBar = {
            LibraryTopBar(
                songCount = songCount,
                totalDuration = totalDuration,
                totalSize = totalSize,
                onSortClick = { showSortDialog = true },
                onRefresh = { viewModel.refreshLibrary() }
            )
        },
        floatingActionButton = {
            if (!isScanning) {
                FloatingActionButton(
                    onClick = { viewModel.refreshLibrary() },
                    containerColor = MaterialTheme.colorScheme.primary
                ) {
                    Icon(Icons.Default.Refresh, contentDescription = "Refresh")
                }
            }
        }
    ) { paddingValues ->
        Box(modifier = Modifier.padding(paddingValues)) {
            if (isScanning && songs.isEmpty()) {
                // Full-screen shimmer loading
                ShimmerLibraryList()
            } else {
                LazyColumn(
                    modifier = Modifier.fillMaxSize(),
                    contentPadding = PaddingValues(bottom = if (playerState != null) 80.dp else 16.dp)
                ) {
                    itemsIndexed(
                        items = songs,
                        key = { _, song -> song.filePath }
                    ) { index, song ->
                        AnimatedSongItem(
                            song = song,
                            onPlay = { onPlaySong(song) },
                            onToggleFavorite = { viewModel.toggleFavorite(song) },
                            onDelete = { viewModel.deleteSong(song) },
                            modifier = Modifier.animateItemPlacement()
                        )
                        if (index < songs.lastIndex) Divider()
                    }
                }
            }

            // Scanning progress overlay
            if (isScanning) {
                LinearProgressIndicator(
                    progress = if (scanProgress.second > 0) scanProgress.first.toFloat() / scanProgress.second else 0f,
                    modifier = Modifier
                        .fillMaxWidth()
                        .align(Alignment.TopCenter)
                )
            }
        }
    }

    // Sort dialog
    if (showSortDialog) {
        SortOrderDialog(
            currentOrder = sortOrder,
            onOrderSelected = {
                viewModel.updateSortOrder(it)
                showSortDialog = false
            },
            onDismiss = { showSortDialog = false }
        )
    }
}

@Composable
fun AnimatedSongItem(
    song: LocalSong,
    onPlay: () -> Unit,
    onToggleFavorite: () -> Unit,
    onDelete: () -> Unit,
    modifier: Modifier = Modifier
) {
    var showDeleteConfirm by remember { mutableStateOf(false) }

    SwipeToDismiss(
        state = rememberDismissState(
            confirmValueChange = { dismissValue ->
                if (dismissValue == DismissValue.DismissedToEnd) {
                    onPlay()
                } else if (dismissValue == DismissValue.DismissedToStart) {
                    showDeleteConfirm = true
                }
                false // We don't want automatic dismissal
            }
        ),
        background = {
            val color = when (dismissState.dismissDirection) {
                DismissDirection.StartToEnd -> Color(0xFF4CAF50)
                DismissDirection.EndToStart -> Color(0xFFF44336)
                null -> Color.Transparent
            }
            Box(
                modifier = Modifier
                    .fillMaxSize()
                    .background(color)
                    .padding(horizontal = 20.dp),
                contentAlignment = when (dismissState.dismissDirection) {
                    DismissDirection.StartToEnd -> Alignment.CenterStart
                    DismissDirection.EndToStart -> Alignment.CenterEnd
                    null -> Alignment.Center
                }
            ) {
                Icon(
                    imageVector = when (dismissState.dismissDirection) {
                        DismissDirection.StartToEnd -> Icons.Default.PlayArrow
                        DismissDirection.EndToStart -> Icons.Default.Delete
                        null -> Icons.Default.PlayArrow
                    },
                    contentDescription = null,
                    tint = Color.White
                )
            }
        },
        dismissContent = {
            ListItem(
                headlineContent = { Text(song.title, maxLines = 1, overflow = TextOverflow.Ellipsis) },
                supportingContent = { Text("${song.artist} • ${song.album}") },
                leadingContent = {
                    // Cover art image (could be loaded with Coil/Glide)
                    Box(
                        modifier = Modifier
                            .size(48.dp)
                            .clip(RoundedCornerShape(8.dp))
                            .background(MaterialTheme.colorScheme.secondaryContainer)
                    ) {
                        if (song.coverArtPath != null) {
                            Image(
                                painter = rememberImagePainter(File(song.coverArtPath)),
                                contentDescription = null,
                                modifier = Modifier.fillMaxSize(),
                                contentScale = ContentScale.Crop
                            )
                        } else {
                            Icon(
                                Icons.Default.MusicNote,
                                contentDescription = null,
                                modifier = Modifier.align(Alignment.Center)
                            )
                        }
                    }
                },
                trailingContent = {
                    Row {
                        IconButton(onClick = onToggleFavorite) {
                            Icon(
                                if (song.isFavorite) Icons.Filled.Favorite else Icons.Default.FavoriteBorder,
                                contentDescription = "Favorite",
                                tint = if (song.isFavorite) Color.Red else MaterialTheme.colorScheme.onSurfaceVariant
                            )
                        }
                        Text(
                            text = formatDuration(song.duration),
                            style = MaterialTheme.typography.bodySmall,
                            modifier = Modifier.align(Alignment.CenterVertically)
                        )
                    }
                },
                modifier = modifier
                    .fillMaxWidth()
                    .clickable { onPlay() }
                    .padding(horizontal = 16.dp, vertical = 4.dp)
            )
        }
    )

    if (showDeleteConfirm) {
        AlertDialog(
            onDismissRequest = { showDeleteConfirm = false },
            title = { Text("Delete Song") },
            text = { Text("Are you sure you want to delete '${song.title}'?") },
            confirmButton = {
                TextButton(onClick = {
                    onDelete()
                    showDeleteConfirm = false
                }) {
                    Text("Delete")
                }
            },
            dismissButton = {
                TextButton(onClick = { showDeleteConfirm = false }) {
                    Text("Cancel")
                }
            }
        )
    }
}
```

Example shimmer modifier:

```kotlin
fun Modifier.shimmerEffect(): Modifier = composed {
    val transition = rememberInfiniteTransition()
    val alpha by transition.animateFloat(
        initialValue = 0.2f, targetValue = 0.6f,
        animationSpec = infiniteRepeatable(tween(1000), RepeatMode.Reverse)
    )
    background(Color.LightGray.copy(alpha = alpha))
}
```
